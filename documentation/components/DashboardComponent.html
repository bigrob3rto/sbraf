import { Component, OnInit, ViewChild } from '@angular/core';
import { Printable } from '../../prod-table/printable.component';
import { DateRangeComponent } from '../../views/daterange/daterange.component';
import { EconomicComponent } from './economic/economic.component';
import { PerformanceComponent } from './performance/performance.component';



@Component({
  templateUrl: 'dashboard.component.html',
  styleUrls: ['./dashboard.component.css']
})
export class DashboardComponent extends Printable {
  public loading: boolean = false;
  public title = "Dashboard"
  public active; // = 2;

  btnRefresh = true;

  // @ViewChild(DateRangeComponent, { static: false }) daterangeComponent: DateRangeComponent;
  @ViewChild('daterange') daterange: DateRangeComponent;
  @ViewChild('economic') economic: EconomicComponent;
  @ViewChild('performance') performance: PerformanceComponent;


  label = {
    1: "Economic Trend",
    2: 'Menu Performance',
    3: 'Forecasting Section'
  }

  constructor() {
    super();
  }



  setLoading(value) {
    this.loading = value;
    // console.log("Received", value)
  }

  refresh = () => {
    if (this.active == 1)
      this.economic.refresh();
    if (this.active == 2)
      this.performance.refresh();
  }

  printPrepare(v) {
    if (this.active == 2)
      this.performance.printPrepare(v);
  };

}
