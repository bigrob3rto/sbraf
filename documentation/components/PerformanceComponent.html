import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { NgbDateParserFormatter } from '@ng-bootstrap/ng-bootstrap';
import { TableLib } from '../../../prod-table/lib/table-lib';
import { AwsService } from '../../../service/aws.service';
import { MessageService } from '../../../service/message.service';
import { DateRangeComponent } from '../../../views/daterange/daterange.component';

import { Order } from '../../../prod-table/order/order-table.component';
import { ColumnApi, GridApi } from 'ag-grid-community';
import { Category } from '../../../prod-table/cat-table.component';
import { Product } from '../../../prod-table/prod-table.component';



@Component({
  selector: 'app-performance',
  templateUrl: 'performance.component.html',
  styleUrls: ['./performance.component.css']
})
export class PerformanceComponent implements OnInit {
  public gridApi: GridApi;
  public gridColumnApi: ColumnApi;
  public gridApi_left: GridApi;
  public gridColumnApi_left: ColumnApi;
  public rowData = [];
  public domLayout = 'autoHeight';
  public theme;

  @Input() daterange: DateRangeComponent;
  @Output() loading = new EventEmitter<boolean>();



  public columnDefs_left = [
    {
      headerName: 'Category', rowGroup: true, field: 'category', sortable: true, resizable: true, width: 150
    },
    {
      headerName: 'Product', field: 'product', sortable: true, resizable: true, width: 400
    },
    {
      headerName: 'Sold', field: 'sold', sortable: true, resizable: true, width: 150, sort: 'desc',
    },
    {
      headerName: 'Total Gross', field: 'gross', sortable: true, resizable: true, width: 150
    },
  ];

  public columnDefs = [
    {
      headerName: 'Category', field: 'category', sortable: true, resizable: true, width: 150
    },
    {
      headerName: 'Product', field: 'product', sortable: true, resizable: true, width: 400
    },
    {
      headerName: 'Sold', field: 'sold', sortable: true, resizable: true, width: 150, sort: 'desc',
    },
    {
      headerName: 'Total Gross', field: 'gross', sortable: true, resizable: true, width: 150
    },
  ];

  constructor(
    protected awsService: AwsService,
    protected messageService: MessageService,
    protected ngbDateParserFormatter: NgbDateParserFormatter,
  ) {
    // load theme from local storage
    this.theme = localStorage.getItem('ag-grid-theme');
  }

  /*************************************** */
  ngOnInit() {
    setTimeout(() => {
      this.refresh();
    }, 0);
  }

  // extend to avoid inital refresh
  onGridReady(params) {
    this.gridApi = params.api;
    this.gridColumnApi = params.columnApi;
  }

  // extend to avoid inital refresh
  onGridReady_left(params) {
    this.gridApi_left = params.api;
    this.gridColumnApi_left = params.columnApi;
  }

  onGridSizeChanged(params) {
    params.api.sizeColumnsToFit();
  }

  /*************************************** */
  async refresh() {
    // set loading flag
    this.loading.emit(true);

    let start = this.ngbDateParserFormatter.format(this.daterange.date1); // e.g. myDate = 2017-01-01
    let stop = this.ngbDateParserFormatter.format(this.daterange.date2); // e.g. myDate = 2017-01-01

    // whole period statistics
    const products: Product[] = await this.awsService.getProducts();
    // trasform category ID into catagory name
    let catArr: Category[] = await this.awsService.getCategories();
    // load orders in period
    const orders_all: Order[] = await this.awsService.getAllOrdersV2(start, stop);

    // get products ordered
    let o_products = orders_all.map(o => o.product_id);
    // filter unique values
    o_products = o_products.filter((x, i, a) => a.indexOf(x) == i);
    o_products.forEach(op => {
      const product = products.find(p => p.product_id == op);

      let catFound;
      if (product) {
        const category_id = product.product_category_id;
        catFound = catArr.find(x => x.categoryid == String(category_id));  // find corresponding category
      }

      let row = {
        category: catFound ? catFound.name : '',
        product: product ? product.product_name : '',
        sold: orders_all.reduce((sum, o) => sum + (o.product_id == op ? o.quantity : 0), 0),
        gross: TableLib.currencyFormatter(orders_all.reduce((sum, o) => sum + (o.product_id == op ? o.price : 0), 0)),
      }
      this.rowData.push(row);
    })

    this.gridApi.setRowData(this.rowData);
    this.gridApi_left.setRowData(this.rowData);
    // clear loading flag
    this.loading.emit(false);
  }


  /*********************************** 
   * Extends parent function
  */
  public printPrepare(value: boolean) {
    if (value) {
      this.gridApi_left.setDomLayout('print');
      this.gridApi.setDomLayout('print');
    }
    else {
      this.gridApi_left.setDomLayout('autoHeight');
      this.gridApi.setDomLayout('autoHeight');

    }
  }
}
